> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Hebrew RSVP site for Mili Arbibe's birthday at rsvp.arbibe.dev — UUID-per-guest links, SMS OTP via Twilio, admin dashboard with Google auth, hosted on Vercel + Supabase.

**Architecture:** Next.js 14 App Router on Vercel. Server components fetch data; client components handle interactive UI (RSVP state machine, admin forms). All DB access via Supabase service role key in API routes. Guest verification: OTP → session token stored in `guests.session_token`. Admin protected by NextAuth Google OAuth with email whitelist.

**Tech Stack:** Next.js 14, TypeScript, Tailwind CSS v3, @supabase/supabase-js v2, next-auth v4, twilio v5, Vercel, Supabase PostgreSQL

```typescript
  const cleaned = raw.replace(/[\s\-\(\)]/g, '')
  if (cleaned.startsWith('+972')) return cleaned
  if (cleaned.startsWith('972')) return '+' + cleaned
```

- [ ] **Step 2: Create app/api/otp/send/route.ts**

```typescript
import { NextRequest, NextResponse } from 'next/server'
import { randomInt } from 'crypto'
import { supabaseAdmin } from '@/lib/supabase'
import { sendOtpSms, normalizeIsraeliPhone } from '@/lib/twilio'

export async function POST(req: NextRequest) {
  let body: { guestId?: string; phone?: string }
  try {
    body = await req.json()
  } catch {
    return NextResponse.json({ error: 'Invalid JSON' }, { status: 400 })
  }

  const { guestId, phone } = body
  if (!guestId || !phone) {
    return NextResponse.json({ error: 'guestId and phone required' }, { status: 400 })
  }

  // Verify guest exists and is still pending
  const { data: guest } = await supabaseAdmin
    .from('guests')
    .select('id, rsvp_status')
    .eq('id', guestId)
    .single()

  if (!guest) {
    return NextResponse.json({ error: 'Guest not found' }, { status: 404 })
  }
  if (guest.rsvp_status !== 'pending') {
    return NextResponse.json({ error: 'Already responded' }, { status: 400 })
  }

  const normalizedPhone = normalizeIsraeliPhone(phone)

  // Rate limit: max 3 OTP sends per phone per hour
  const { count } = await supabaseAdmin
    .from('otp_codes')
    .select('*', { count: 'exact', head: true })
    .eq('phone', normalizedPhone)
    .gte('created_at', new Date(Date.now() - 60 * 60 * 1000).toISOString())

  if ((count ?? 0) >= 3) {
    return NextResponse.json(
      { error: 'יותר מדי ניסיונות. נסה שוב בעוד שעה.' },
      { status: 429 }
    )
  }

  const code = String(randomInt(100000, 999999))
  const expiresAt = new Date(Date.now() + 10 * 60 * 1000).toISOString()

  await supabaseAdmin.from('otp_codes').insert({
    guest_id: guestId,
    phone: normalizedPhone,
    code,
    expires_at: expiresAt,
  })

  await sendOtpSms(normalizedPhone, code)

  return NextResponse.json({ success: true })
}
```

- [ ] **Step 3: Create app/api/otp/verify/route.ts**

```typescript
import { NextRequest, NextResponse } from 'next/server'
import { randomUUID } from 'crypto'
import { supabaseAdmin } from '@/lib/supabase'
import { normalizeIsraeliPhone } from '@/lib/twilio'

export async function POST(req: NextRequest) {
  let body: { guestId?: string; phone?: string; code?: string }
  try {
    body = await req.json()
  } catch {
    return NextResponse.json({ error: 'Invalid JSON' }, { status: 400 })
  }

  const { guestId, phone, code } = body
  if (!guestId || !phone || !code) {
    return NextResponse.json({ error: 'guestId, phone, and code required' }, { status: 400 })
  }

  const normalizedPhone = normalizeIsraeliPhone(phone)
  const now = new Date().toISOString()

  const { data: otp } = await supabaseAdmin
    .from('otp_codes')
    .select('id')
    .eq('guest_id', guestId)
    .eq('phone', normalizedPhone)
    .eq('code', code)
    .eq('used', false)
    .gte('expires_at', now)
    .order('created_at', { ascending: false })
    .limit(1)
    .single()

  if (!otp) {
    return NextResponse.json({ error: 'קוד שגוי או שפג תוקפו' }, { status: 400 })
  }

  // Mark used
  await supabaseAdmin.from('otp_codes').update({ used: true }).eq('id', otp.id)

  // Issue session token + store phone
  const sessionToken = randomUUID()
  await supabaseAdmin
    .from('guests')
    .update({ phone: normalizedPhone, session_token: sessionToken })
    .eq('id', guestId)

  return NextResponse.json({ sessionToken })
}
```

```typescript
import { NextRequest, NextResponse } from 'next/server'
import { supabaseAdmin } from '@/lib/supabase'

export async function POST(req: NextRequest) {
  let body: { guestId?: string; sessionToken?: string; attending?: boolean; guestCount?: number }
  try {
    body = await req.json()
  } catch {
    return NextResponse.json({ error: 'Invalid JSON' }, { status: 400 })
  }

  const { guestId, sessionToken, attending, guestCount } = body
  if (!guestId || !sessionToken || attending === undefined) {
    return NextResponse.json({ error: 'Missing required fields' }, { status: 400 })
  }
  if (attending && (!guestCount || guestCount < 1)) {
    return NextResponse.json({ error: 'guestCount required when attending' }, { status: 400 })
  }

  // Verify session token
  const { data: guest } = await supabaseAdmin
    .from('guests')
    .select('id, rsvp_status')
    .eq('id', guestId)
    .eq('session_token', sessionToken)
    .single()

  if (!guest) {
    return NextResponse.json({ error: 'Unauthorized' }, { status: 401 })
  }
  if (guest.rsvp_status !== 'pending') {
    return NextResponse.json({ error: 'Already responded' }, { status: 400 })
  }

  await supabaseAdmin
    .from('guests')
    .update({
      rsvp_status: attending ? 'confirmed' : 'declined',
      guest_count: attending ? guestCount : null,
      responded_at: new Date().toISOString(),
      session_token: null,
    })
    .eq('id', guestId)

  return NextResponse.json({ success: true })
}
```

```typescript
'use client'
import { useState, useEffect } from 'react'

interface Props {
  guestId: string
  phone: string
  onSuccess: (sessionToken: string) => void
  onResend: () => void
}

export default function OtpStep({ guestId, phone, onSuccess }: Props) {
  const [code, setCode] = useState('')
  const [loading, setLoading] = useState(false)
  const [error, setError] = useState('')
  const [attempts, setAttempts] = useState(0)
  const [resendCooldown, setResendCooldown] = useState(60)
  const [resending, setResending] = useState(false)

  useEffect(() => {
    if (resendCooldown <= 0) return
    const t = setTimeout(() => setResendCooldown((c) => c - 1), 1000)
    return () => clearTimeout(t)
  }, [resendCooldown])

  async function handleSubmit(e: React.FormEvent) {
    e.preventDefault()
    if (attempts >= 3) return
    setError('')
    setLoading(true)

    const res = await fetch('/api/otp/verify', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ guestId, phone, code }),
    })

    const data = await res.json()
    setLoading(false)

    if (!res.ok) {
      const newAttempts = attempts + 1
      setAttempts(newAttempts)
      if (newAttempts >= 3) {
        setError('חרגת ממספר הניסיונות. רפרש את הדף ונסה שוב.')
      } else {
        setError(data.error ?? 'קוד שגוי')
      }
      return
    }

    onSuccess(data.sessionToken)
  }

  async function handleResend() {
    setResending(true)
    setError('')
    await fetch('/api/otp/send', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ guestId, phone }),
    })
    setResending(false)
    setResendCooldown(60)
    setCode('')
  }

  return (
    <div className="min-h-screen bg-cream flex flex-col items-center justify-center p-6">
      <div className="w-full max-w-sm bg-white rounded-3xl shadow-sm p-8">
        <h2 className="font-serif text-2xl text-center mb-2">הזן קוד אימות</h2>
        <p className="text-center text-charcoal/60 text-sm mb-2">
          שלחנו קוד ל-{phone}
        </p>

        <form onSubmit={handleSubmit} className="space-y-4 mt-6">
          <input
            type="text"
            inputMode="numeric"
            pattern="[0-9]{6}"
            maxLength={6}
            dir="ltr"
            placeholder="123456"
            value={code}
            onChange={(e) => setCode(e.target.value.replace(/\D/g, ''))}
            autoComplete="one-time-code"
            className="w-full border border-blush rounded-2xl px-4 py-3 text-center text-2xl tracking-widest font-mono focus:outline-none focus:ring-2 focus:ring-blush"
            required
            disabled={attempts >= 3}
          />

          {error && <p className="text-red-500 text-sm text-center">{error}</p>}

          <button
            type="submit"
            disabled={loading || attempts >= 3 || code.length !== 6}
            className="w-full bg-blush text-charcoal font-semibold py-3 rounded-full hover:opacity-90 transition disabled:opacity-50"
          >
            {loading ? 'מאמת...' : 'אמת קוד'}
          </button>
        </form>

        <div className="mt-4 text-center">
          {resendCooldown > 0 ? (
            <p className="text-charcoal/40 text-sm">שלח מחדש בעוד {resendCooldown}ש׳</p>
          ) : (
            <button
              onClick={handleResend}
              disabled={resending}
              className="text-charcoal/60 text-sm underline hover:text-charcoal disabled:opacity-50"
            >
              {resending ? 'שולח...' : 'שלח קוד מחדש'}
            </button>
          )}
        </div>
      </div>
    </div>
  )
}
```

```typescript
import { NextRequest, NextResponse } from 'next/server'
import { getServerSession } from 'next-auth'
import { authOptions } from '@/lib/auth'
import { supabaseAdmin } from '@/lib/supabase'

async function requireAdmin() {
  const session = await getServerSession(authOptions)
  if (!session) return null
  return session
}

export async function GET() {
  const session = await requireAdmin()
  if (!session) return NextResponse.json({ error: 'Unauthorized' }, { status: 401 })

  const { data: guests, error } = await supabaseAdmin
    .from('guests')
    .select('*')
    .order('created_at', { ascending: true })

  if (error) return NextResponse.json({ error: error.message }, { status: 500 })
  return NextResponse.json(guests ?? [])
}

export async function POST(req: NextRequest) {
  const session = await requireAdmin()
  if (!session) return NextResponse.json({ error: 'Unauthorized' }, { status: 401 })

  let body: { name?: string; phone?: string }
  try {
    body = await req.json()
  } catch {
    return NextResponse.json({ error: 'Invalid JSON' }, { status: 400 })
  }

  if (!body.name?.trim()) {
    return NextResponse.json({ error: 'Name required' }, { status: 400 })
  }

  const { data: guest, error } = await supabaseAdmin
    .from('guests')
    .insert({ name: body.name.trim(), phone: body.phone?.trim() || null })
    .select()
    .single()

  if (error) return NextResponse.json({ error: error.message }, { status: 500 })
  return NextResponse.json(guest, { status: 201 })
}
```

```typescript
'use client'
import { useState } from 'react'
import type { Guest } from '@/types'

interface Props {
  onAdded: (guest: Guest) => void
}

export default function AddGuestForm({ onAdded }: Props) {
  const [name, setName] = useState('')
  const [phone, setPhone] = useState('')
  const [loading, setLoading] = useState(false)
  const [error, setError] = useState('')
  const [lastLink, setLastLink] = useState<string | null>(null)
  const [copied, setCopied] = useState(false)

  async function handleSubmit(e: React.FormEvent) {
    e.preventDefault()
    setError('')
    setLastLink(null)
    setLoading(true)

    const res = await fetch('/api/admin/guests', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ name: name.trim(), phone: phone.trim() || undefined }),
    })

    const data = await res.json()
    setLoading(false)

    if (!res.ok) {
      setError(data.error ?? 'שגיאה')
      return
    }

    const link = `${window.location.origin}/rsvp/${data.id}`
    setLastLink(link)
    onAdded(data)
    setName('')
    setPhone('')
  }

  function copyLink() {
    if (!lastLink) return
    navigator.clipboard.writeText(lastLink)
    setCopied(true)
    setTimeout(() => setCopied(false), 2000)
  }

  return (
    <div className="bg-white rounded-2xl shadow-sm p-6 mb-6">
      <h2 className="font-serif text-lg mb-4">הוסף אורח</h2>
      <form onSubmit={handleSubmit} className="flex flex-wrap gap-3 items-end">
        <div className="flex-1 min-w-[140px]">
          <label className="text-xs text-charcoal/60 mb-1 block">שם *</label>
          <input
            type="text"
            value={name}
            onChange={(e) => setName(e.target.value)}
            placeholder="ישראל ישראלי"
            required
            className="w-full border border-blush/50 rounded-xl px-3 py-2 text-sm focus:outline-none focus:ring-2 focus:ring-blush"
          />
        </div>
        <div className="flex-1 min-w-[140px]">
          <label className="text-xs text-charcoal/60 mb-1 block">טלפון (אופציונלי)</label>
          <input
            type="tel"
            dir="ltr"
            value={phone}
            onChange={(e) => setPhone(e.target.value)}
            placeholder="05X-XXXXXXX"
            className="w-full border border-blush/50 rounded-xl px-3 py-2 text-sm focus:outline-none focus:ring-2 focus:ring-blush"
          />
        </div>
        <button
          type="submit"
          disabled={loading}
          className="bg-blush text-charcoal px-5 py-2 rounded-xl font-semibold text-sm hover:opacity-90 transition disabled:opacity-50"
        >
          {loading ? '...' : '+ הוסף'}
        </button>
      </form>

      {error && <p className="text-red-500 text-sm mt-2">{error}</p>}

      {lastLink && (
        <div className="mt-4 flex items-center gap-2 bg-cream rounded-xl p-3">
          <p className="text-xs text-charcoal/60 flex-1 font-mono truncate">{lastLink}</p>
          <button
            onClick={copyLink}
            className="text-xs bg-white px-3 py-1 rounded-lg hover:bg-blush/30 transition whitespace-nowrap"
          >
            {copied ? '✓' : 'העתק'}
          </button>
        </div>
      )}
    </div>
  )
}
```

2. Get a phone number that supports SMS to Israel
3. Copy from Console:
   - `TWILIO_ACCOUNT_SID`
   - `TWILIO_AUTH_TOKEN`
   - `TWILIO_PHONE_NUMBER` (your Twilio number, e.g. +19876543210)
